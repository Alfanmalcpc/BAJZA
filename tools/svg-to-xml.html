document.getElementById('copyBtn').addEventListener('click', () => {
  if (!outputCode.value) return;
  navigator.clipboard.writeText(outputCode.value).then(() => showToast('\u2705 XML Copied!'));
});

document.getElementById('downloadBtn').addEventListener('click', () => {
  if (!outputCode.value) return;
  const blob = new Blob([outputCode.value], { type: 'application/xml' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  let filename = 'vector.xml';
  if (fileInput.files.length > 0) filename = fileInput.files[0].name.replace('.svg', '.xml');
  a.download = filename;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);
  showToast('\ud83d\udcbe Downloading XML...');
});

document.getElementById('clearBtn').addEventListener('click', () => {

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2500);
}
